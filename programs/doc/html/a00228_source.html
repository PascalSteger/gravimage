53 # Launch a simple MPI run, with no communication of covariance matrix
54 #
55 # Each process will make sure to initialise the folder if needed. Then and
56 # only then, it will send the signal to its next in line to proceed. This
57 # allows for initialisation over an arbitrary cluster geometry (you can have
58 # a single node with many cores, and all the chains living there, or many
59 # nodes with few cores). The speed loss due to the time spend checking if the
60 # folder is created should be negligible when running decently sized chains.
61 #
62 # Each process will send the number that it found to be the first available
63 # to its friends, so that the gathering of information post-run is made
64 # easier. If a chain number is specified, this will be used as the first
65 # number, and then incremented afterwards with the rank of the process.
66 #
67 def mpi_run(custom_command=""):
68 
69  from mpi4py import MPI
70 
71  comm = MPI.COMM_WORLD
72  nprocs = comm.Get_size()
73  rank = comm.Get_rank()
74 
75  success = True
76  folder = ''
77 
78  # If the process is not the zeroth one, then wait for a signal from your
79  # n-1 before initializing the folder
80  if rank != 0:
81  status = comm.recv(source=rank-1, tag=1)
82  folder = comm.recv(source=rank-1, tag=2)
83  if status == 'failed':
84  success = False
85  else:
86  number = status
87 
88  if success:
89  if not custom_command:
90  custom_command = " ".join(sys.argv[1:])
91  if rank != 0:
92  custom_command += " --chain-number %s" % str(int(number)+1)
93 
94  # First check if the folder is there
95  already_sent = False
96  if rank != 0 and rank < nprocs-1:
97  status = int(number)+1
98  if Data.folder_is_initialised(folder):
99  comm.send(status, dest=rank+1, tag=1)
100  comm.send(folder, dest=rank+1, tag=2)
101  already_sent = True
102 
103  # Then, properly initialise
104  cosmo, data, command_line, success = safe_initialisation(
105  custom_command, comm, nprocs)
106 
107  # The first initialisation should check a few more things
108  if rank == 0:
109  # Check that the run asked is compatible with mpirun and prepare.
110  if command_line.subparser_name == 'info':
111  warnings.warn(
112  "Analyzing the chains is not supported in mpirun"
113  " so this will run on one core only.")
114  status = 'failed'
115  elif command_line.method == "MH":
116  regexp = re.match(".*__(\w*).txt", data.out_name)
117  suffix = regexp.groups()[0]
118  status = suffix
119  elif command_line.method == "NS":
120  status = 1
121  else:
122  warnings.warn(
123  "The method '%s' is not supported"%(command_line.method) +
124  " in mpirun so this will run on one core only.")
125  status = 'failed'

127  elif rank < nprocs-1:
128  status = int(number)+1

130  # number for the other to add 1
131  if rank < nprocs-1:
132  if not already_sent:
133  comm.send(status, dest=rank+1, tag=1)
134  comm.send(folder, dest=rank+1, tag=2)
135  else:

137  comm.send('failed', dest=rank+1, tag=1)
138  comm.send('', dest=rank+1, tag=2)
139

191 def safe_initialisation(custom_command="", comm=None, nprocs=1):
192  try:
193  cosmo, data, command_line, success = initialise(custom_command)
194  except io_mp.ConfigurationError as message:
195  if comm:
196  for index in range(1, nprocs):
197  comm.send('failed', dest=index, tag=1)
198  print str(message)
199  raise io_mp.ConfigurationError(
200  "The initialisation was not successful, resulting in a "
201  "potentially half created `log.param`. Please see the "
202  "above error message. If you run the exact same command, it"
203  " will not work. You should solve the problem, and try again.")
204  except KeyError as e:
205  if comm:
206  for index in range(1, nprocs):
207  comm.send('failed', dest=index, tag=1)
208  raise io_mp.ConfigurationError(
209  "You are running in a folder that was created following "
210  "a non-successful initialisation (wrong parameter name, "
211  "wrong likelihood, etc...). If you have solved the issue, you "
212  "should remove completely the output folder, and try again." +
213  " Alternatively, there could be a problem with "+e.message)
214  return cosmo, data, command_line, success
