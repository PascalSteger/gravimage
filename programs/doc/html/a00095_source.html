131  # Wolf, Walker method for M_half, r_half
132  # assuming isotropic Plummer profile:
133  r_half = gp.dat.rhalf[0] # [pc] from overall rho*
134  sigv = max(gp.dat.sig[0]) # [km/s] from overall rho*
135  M_half_walk = 5.*r_half*sigv**2/(2.*gu.G1__pcMsun_1km2s_2) # [Munit] Walker Penarrubia 2011
136  r_half_walk = r_half
137  # other estimate: Wolf+2010,

139  r_half_wolf = 4/3. * r_half
140 
141  # density at half-light radius of baryons
