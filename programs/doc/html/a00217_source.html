25 # Performing the Importance Sampling
26 #
27 # The idea is to start from an existing run, constraining a certain model I,
28 # given a set of experiments. The new run will constrain the same model I,
29 # but adding one or several new experiments. In the case where it is expected
30 # that the final posterior distribution should not differ too greatly between
31 # the two parameter extractions, then using Importance Sampling can speed up
32 # significantly the second one.
33 #
34 # Instead of properly sampling randomly the parameter space, it instead reads
35 # the chains from the previous run, recompute the cosmology at this point,
36 # then adds the log-likelihood contributed by the new experiments to the
37 # previous ones. As an input of the method, with the flag
38 # `--IS-starting-folder`, you can thus specify either a folder containing a
39 # Monte Python run, or a set of chains that you want to be converted.
40 #
41 # The code will automatically compute the minimum amount of things. For
42 # instance, if the first run had all the Planck likelihoods, and the second,
43 # all the Planck likelihoods plus a prior on :math:`H_0`, it would be absurd
44 # to recompute also the cosmological perturbations: the only needed quantity
45 # is a background quantity.
46 #
47 # The new chains will hence store the same points in parameter space, but
48 # with a different value of the likelihood, and also of the multiplicity -
49 # that will become non-integer. Indeed, the multiplicity is also a probe of
50 # the posterior, and this new, higher likelihood should have had a higher
51 # multiplicity.
52 #
53 def run(cosmo, data, command_line):
54  # Check that the command_line "--IS-starting-folder" points to an existing
55  # Monte Python folder run, or a subset of files, and store in any case all
56  # the chains to analyze in the chains.
57  starting_folder = command_line.IS_starting_folder
58  if not starting_folder:
59  raise io_mp.ConfigurationError(
60  "When running importance sampling, you should specify a folder or"
61  " a set of chains with the option '--IS-starting-folder'")
62  chains = []
63  # If starting_folder is of length 1, it means it is either a whole folder,
64  # or just one chain. If it is a folder, we recover all chains within.
65  if len(starting_folder) == 1:
66  starting_folder = starting_folder[0]
67  if os.path.isdir(starting_folder):
68  for elem in os.listdir(starting_folder):
69  if elem.find("__") != -1:
70  chains.append(elem)
71  # Else, it is a list of chains, of which we recover folder name, and store
72  # all of them in chains.
73  else:
74  chains = starting_folder
75  starting_folder = os.path.sep.join(chains[0].split(os.path.sep)[:-1])
76  chains = [elem.split(os.path.sep)[-1] for elem in chains]
77 
78  # Recovering only the extra likelihoods
79  new_experiments = recover_new_experiments(
80  data, command_line, starting_folder)
81  if not new_experiments:
82  # it means that some experiments that were used before are no longer
83  # used. In this case, one should recompute all the likelihoods.
84  new_experiments = data.experiments
85 
86  # resetting the needed cosmo arguments, and deleting the dictionary of
87  # likelihoods, only if new_experiments is smaller than the old ones.
88  ignore_likelihood = True
89  if set(new_experiments) < set(data.experiments):
90  data.cosmo_arguments = {}
91  data.initialise_likelihoods(new_experiments)
92  ignore_likelihood = False
93  elif set(new_experiments) == set(data.experiments):
94  warnings.warn(
95  "All likelihoods were found to be different than ones from the "
96  "starting folder, or some were not used any longer. The previous "
97  "value of the likelihood will be discarded.")
98  else:
99  io_mp.ConfigurationError(
100  "You probably tried to run Importance Sampling with less "
101  "experiments than what you started with. This is not supported.")
102 
103  # Multiprocessing part, to analyze all the chains in parallel. When not
104  # specifying any 'processes' keyword argument to the Pool call, the system
105  # uses as many as possible.
106  pool = Pool()
107  args = [(data, cosmo, command_line, starting_folder,
108  elem, ignore_likelihood) for elem in chains]
109  # Note the use of translate_chain_star, and not translate_chain, because of
110  # the limitations of the `map` function (it only takes one argument). The
111  # `_star` function simply unwraps the argument.
112  print '\nStart extracting the chains:\n'
113  pool.map(translate_chain_star, args)
114  # Close the pool, and join everything (the join might not be needed)
115  pool.close()
116  pool.join()

124 def recover_new_experiments(data, command_line, starting_folder):
125  # Initialize the companion data structure, on a modified command line

127  modified_command_line.folder = starting_folder
128  modified_command_line.param = os.path.join(starting_folder, 'log.param')
129 
130  # Go through the file, and stop when you find the good line. The previous
131  # way of doing, to simply initialise another data instance fails when using
132  # Planck. Indeed, clik likelihoods can not be initialised twice.
133  print 'Reading the starting folder'
134  print '---------------------------'

150 # Translate the input to the output
151 #
152 # .. note::
153 #
154 # If the keyword argument `ignore_likelihood` is set to true, the
155 # previous value of the likelihood is discarded.
156 #
157 #
158 def translate_chain(data, cosmo, command_line,
159  starting_folder, chain_name, ignore_likelihood=False):
160 
161  input_path = os.path.join(starting_folder, chain_name)
162  output_path = os.path.join(command_line.folder, chain_name)
163  print ' -> reading ', input_path
164  parameter_names = data.get_mcmc_parameters(['varying'])
165  with open(input_path, 'r') as input_chain:
166  with open(output_path, 'w') as output_chain:
167  for line in input_chain:
168  params = line.split()
169  # recover the likelihood of this point
170  if not ignore_likelihood:
171  loglike = -float(params[1])
172  else:
173  loglike = 0
174  N = float(params[0])
175  # Assign all the recovered values to the data structure
176  for index, param in enumerate(parameter_names):
177  data.mcmc_parameters[param]['current'] = \
178  float(params[2+index])
179  data.update_cosmo_arguments()
180 
181  newloglike = sampler.compute_lkl(cosmo, data)
182 
183  weight = math.exp(newloglike)
184  newloglike += loglike
185  # Accept the point
186  sampler.accept_step(data)
187  io_mp.print_vector([output_chain], N*weight, newloglike, data)
188  print output_path, 'written'
