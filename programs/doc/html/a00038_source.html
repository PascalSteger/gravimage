155 def get_rhohalfs(gp):

157  # Wolf, Walker method for M_half, r_half
158  # assuming isotropic Plummer profile:
159  r_half = gp.dat.rhalf[1] # [pc] from first population
160  sigv = max(gp.dat.sig[1]) # [km/s] from first population
161  M_half_walk = 5.*r_half*sigv**2/(2.*gu.G1__pcMsun_1km2s_2) # [Munit] Walker Penarrubia 2011
162  r_half_walk = r_half
163  # other estimate: Wolf+2010,

166 
167  # density at half-light radius of baryons
