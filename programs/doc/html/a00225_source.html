200 #
201 # Main call to run the MultiNest sampler.
202 #
203 # Note the unusual set-up here, with the two following functions, `prior` and
204 # `loglike` having their docstrings written in the encompassing function.
205 # This trick was necessary as MultiNest required these two functions to be
206 # defined with a given number of parameters, so we could not add `data`. By
207 # defining them inside the run function, this problem was by-passed.
208 #
209 # .. function:: prior
210 #
211 # Generate the prior function for MultiNest
212 #
213 # It should transform the input unit cube into the parameter cube. This
214 # function actually wraps the method :func:`map_from_unit_interval()
215 # <prior.Prior.map_from_unit_interval>` of the class :class:`Prior
216 # <prior.Prior>`.
217 #
218 # Parameters
219 # ----------
220 # cube : array
221 # Contains the current point in unit parameter space that has been
222 # selected within the MultiNest part.
223 # ndim : int
224 # Number of varying parameters
225 # nparams : int
226 # Total number of parameters, including the derived ones (not used,
227 # so hidden in `*args`)
228 #
229 #
230 # .. function:: loglike
231 #
232 # Generate the Likelihood function for MultiNest
233 #
234 # Parameters
235 # ----------
236 # cube : array
237 # Contains the current point in the correct parameter space after
238 # transformation from :func:`prior`.
239 # ndim : int
240 # Number of varying parameters
241 # nparams : int
242 # Total number of parameters, including the derived ones (not used,
243 # so hidden in `*args`)
244 #
245 #
246 def run(cosmo, data, command_line):
247  # Convenience variables
248  derived_param_names = data.get_mcmc_parameters(['derived'])
249  NS_param_names = data.NS_param_names
250 
251  # Function giving the prior probability
252  ##
253  #
254  # Please see the encompassing function docstring
255  #
256  #
257  def prior(cube, ndim, *args):
258  for i, name in zip(range(ndim), NS_param_names):
259  cube[i] = data.mcmc_parameters[name]['prior']\
260  .map_from_unit_interval(cube[i])
261 
262  # Function giving the likelihood probability
263  ##
264  #
265  # Please see the encompassing function docstring
266  #
267  #
268  def loglike(cube, ndim, *args):
269  # Updates values: cube --> data
270  for i, name in zip(range(ndim), NS_param_names):
271  data.mcmc_parameters[name]['current'] = cube[i]
272  # Propagate the information towards the cosmo arguments
273  data.update_cosmo_arguments()
274  lkl = sampler.compute_lkl(cosmo, data)
275  for i, name in enumerate(derived_param_names):
276  cube[ndim+i] = data.mcmc_parameters[name]['current']
277  return lkl
278 
279  # Launch MultiNest, and recover the output code
280  output = nested_run(loglike, prior, **data.NS_arguments)
281 
282  # Assuming this worked, i.e. if output is `None`,
283  # state it and suggest the user to analyse the output.
284  if output is None:
285  warnings.warn('The sampling with MultiNest is done.\n' +
286  'You can now analyse the output calling Monte Python ' +
287  ' with the -info flag in the chain_name/NS subfolder,' +
288  'or, if you used multimodal sampling, in the ' +
289  'chain_name/mode_# subfolders.')

297 #
298 # If mode separation has been performed (i.e., multimodal=True), it creates
