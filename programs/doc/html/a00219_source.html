254 #
255 # Simplistic tex name transformer.
256 #
257 # Essentially tries to add a backslash in front of known possible greek
258 # letters, and insert curly brackets { } around statement following an
259 # _ or a ^. It will also try to include the scale into the name in a nice
260 # way.
261 #
262 # .. note::
263 #
264 # This might easily fail on simple names, like `beta_plus_lambda`. In
265 # this case, please use an extra plot file with the command line option
266 # :code:`-extra plot_file`, or come up with a better function !
267 #
268 # .. note::
269 #
270 # This function returns immediatly with the unmodified name if it already
271 # contains the LaTeX symbol for math, $.
272 #
273 # Parameters
274 # ----------
275 # name : str
276 # Input name
277 #
278 # Keyword Arguments
279 # -----------------
280 # number : float
281 # Scale
282 #
283 #
284 def get_tex_name(name, number=1):
285  # First, if the name already contains $ signs, returns it unmodified
286  if name.find("$") != -1:
287  return name
288  tex_greek = ['omega', 'tau', 'alpha', 'beta', 'delta', 'nu',
289  'Omega', 'Lambda', 'lambda']
290  for elem in tex_greek:
291  if elem in name:
292  position = name.find(elem)
293  name = name[:position]+"""\\"""+name[position:]
294  if name.find('_') != -1:
295  temp_name = name.split('_')[0]+'_{'
296  for i in range(1, len(name.split('_'))):
297  temp_name += name.split('_')[i]+' '
298  name = temp_name + '}'
299  if number == 1:
300  name = "${0}$".format(name)
301  elif number < 1000 and number > 1:
302  name = "$%0.d~%s$" % (number, name)
303  else:
304  temp_name = "$%0.e%s$" % (number, name)
305  m = re.search(r'(?:\$[0-9]*e\+[0]*)([0-9]*)(.*)', temp_name)
306  sign = '+'
307  if m is None:
308  m = re.search(r'(?:\$[0-9]*e\-[0]*)([0-9]*)(.*)', temp_name)
309  sign = '-'
310  name = '$10^{'+sign+m.groups()[0]+'}'+m.groups()[1]
311  return name

417  ##
418  # Imitates the classic tail command
419  def tail(self, lines_2find=1):
420  self.seek(0, 2) # go to end of file
421  bytes_in_file = self.tell()
422  lines_found, total_bytes_scanned = 0, 0
423  while (lines_2find+1 > lines_found and
424  bytes_in_file > total_bytes_scanned):
425  byte_block = min(1024, bytes_in_file-total_bytes_scanned)
426  self.seek(-(byte_block+total_bytes_scanned), 2)
427  total_bytes_scanned += byte_block
428  lines_found += self.read(1024).count('\n')
429  self.seek(-total_bytes_scanned, 2)
430  line_list = list(self.readlines())
431  return line_list[-lines_2find:]
