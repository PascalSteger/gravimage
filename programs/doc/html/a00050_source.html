94  counts, bins = np.histogram(np.log10(self.chis), bins=np.sqrt(len(self.chis)))
95  mincounts = -1
96  found_max = False; found_min = False
97  for k in range(len(counts)):
98  # flag if we found the first max peak in the chi2 distribution
99  if counts[k] < mincounts and not found_max:
100  found_max = True

103  # and set starting minimum chi2 at highest value, to be changed later on
104  if counts[k] >= mincounts and not found_max:
105  mincounts = counts[k]
106  maxcounts = counts[k]

109  mincounts = counts[k]

111  found_min = True
