159 # Translate the output of the Cosmo Hammer into Monte Python chains
160 #
161 # This function will be called by the module :mod:`analyze`.
162 #
163 def from_CH_output_to_chains(folder):
164 
165  chain_name = [a for a in folder.split(os.path.sep) if a][-2]
166  base_name = os.path.join(folder, chain_name)
167  # Recover the points in parameter space
168  with open(base_name+'.out', 'r') as param_values_file:
169  chains = np.loadtxt(param_values_file)
170  # Recover the associated likelihood (it is the -log likelihood)
171  with open(base_name+'prob.out', 'r') as lkl_values_file:
172  lkl = np.loadtxt(lkl_values_file)
173 
174  # Glue them together, with an additional column of ones, for the
175  # multiplicity. This does not mean that the acceptance rate is one, but
176  # points where the code stayed are duplicated in the file.
177 
178  ## First, reshape the lkl array
179  lkl = np.array([[elem] for elem in lkl])
180 
181  ## Create the array of ones
182  ones = np.array([[1] for _ in range(len(lkl))])
183 
184  ## Concatenate everything and save to file
185  final = np.concatenate((ones, lkl, chains), axis=1)
186  output_folder = os.path.join(folder, '..')
187  output_chain_path = os.path.join(output_folder, name_chain)
188  np.savetxt(output_chain_path, final)
