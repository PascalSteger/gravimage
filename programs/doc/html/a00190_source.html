6 # .. moduleauthor:: Benjamin Audren <benjamin.audren@gmail.com>
7 #
8 # Copied from the original c++ code available at
9 # `this address <http://supernovae.in2p3.fr/sdss_snls_jla/ReadMe.html>`_.
10 # The only modification was to keep the names of the covariance matrices to their
11 # original names instead of going for C00, etc... Here is the conversion from
12 # their original file names to theirs. Note that the convention is used for the
13 # computations afterwards.
14 #
15 # .. code::
16 #
17 # C00 = mag_covmat_file
18 # C11 = stretch_covmat_file
19 # C22 = colour_covmat_file
20 # C01 = mag_stretch_covmat_file
21 # C02 = mag_colour_covmat_file
22 # C12 = stretch_colour_covmat_file
23 #
24 # .. note::
25 #
26 # Since there are a lot of file manipulation involved, the "pandas" library
27 # has to be installed -- it is an 8-fold improvement in speed over numpy, and
28 # a 2-fold improvement over a fast Python implementation. The "numexpr"
29 # library is also needed for doing the fast array manipulations, done with
30 # blas daxpy function in the original c++ code. Both can be installed with
31 # pip (Python package manager) easily.
32 #
33 #

77  # Compute negative log-likelihood (eq.15 Betoule et al. 2014)
78  #
79  #
80  def loglkl(self, cosmo, data):
81  # Recover the distance moduli from CLASS (a size N vector of double
82  # containing the predicted distance modulus for each SN in the JLA
83  # sample, given the redshift of the supernova.)
84  redshifts = self.light_curve_params.zcmb
85  size = redshifts.size
86 
87  moduli = np.empty((size, ))
88  for index, row in self.light_curve_params.iterrows():
89  z_cmb = row['zcmb']
90  moduli[index] = cosmo.luminosity_distance(z_cmb)
91  moduli = 5 * np.log10(moduli) + 25
92 
93  # Convenience variables: store the nuisance parameters in short named
94  # variables
95  alpha = (data.mcmc_parameters['alpha']['current'] *
96  data.mcmc_parameters['alpha']['scale'])
97  beta = (data.mcmc_parameters['beta']['current'] *
98  data.mcmc_parameters['beta']['scale'])
99  M = (data.mcmc_parameters['M']['current'] *
100  data.mcmc_parameters['M']['scale'])
101  Delta_M = (data.mcmc_parameters['Delta_M']['current'] *
102  data.mcmc_parameters['Delta_M']['scale'])
103 
104  # Compute the covariance matrix
105  # The module numexpr is used for doing quickly the long multiplication
106  # of arrays (factor of 3 improvements over numpy). It is used as a
107  # replacement of blas routines cblas_dcopy and cblas_daxpy
108  # For numexpr to work, we need (seems like a bug, but anyway) to create
109  # local variables holding the arrays. This cost no time (it is a simple
110  # pointer assignment)
111  C00, C11, C22 = self.C00, self.C11, self.C22
112  C01, C02, C12 = self.C01, self.C02, self.C12
113  cov = ne.evaluate(
114  "(C00 + alpha**2*C11 + beta**2*C22"
115  "+2.*alpha*C01 -2.*beta*C02 -2.*alpha*beta*C12)")
116 
117  # Compute the residuals (estimate of distance moduli - exact moduli)
118  residuals = np.empty((size,))
119  sn = self.light_curve_params
120  # This operation loops over all supernovae!
121  # Compute the approximate moduli
122  residuals = sn.mb - (
123  M - alpha*sn.x1 + beta*sn.color + Delta_M*(
124  sn.thirdvar > self.scriptmcut))
125  # Remove from the approximate moduli the one computed from CLASS
126  residuals -= moduli
127 
128  # Update the diagonal terms of the covariance matrix with the
129  # statistical error
130  cov += np.diag(sn.dmb**2 + (alpha*sn.dx1)**2 + (beta*sn.dcolor)**2
131  + 2.*alpha*sn.cov_m_s
132  - 2.*beta*sn.cov_m_c
133  - 2.*alpha*beta*sn.cov_s_c)
134 
135  # Whiten the residuals, in two steps
136  # 1) Compute the Cholesky decomposition of the covariance matrix, in
137  # place. This is a time expensive (0.015 seconds) part
138  cov = la.cholesky(cov, lower=True, overwrite_a=True)
139 
140  # 2) Solve the triangular system, also time expensive (0.02 seconds)
141  residuals = la.solve_triangular(cov, residuals, lower=True, check_finite=False)
142 
143  # Finally, compute the chi2 as the sum of the squared residuals
144  chi2 = (residuals**2).sum()
145 
146  return -0.5 * chi2
