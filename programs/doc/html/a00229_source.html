556 def compute_fisher(data, cosmo, center, step_size):
557 
558  parameter_names = data.get_mcmc_parameters(['varying'])
559  fisher_matrix = np.zeros(
560  (len(parameter_names), len(parameter_names)), 'float64')
561  # Initialise the gradient field
562  gradient = np.zeros(len(parameter_names), 'float64')
563 
564  for k, elem_k in enumerate(parameter_names):
565  kdiff = center[elem_k]*step_size
566  if kdiff == 0.0:
567  kdiff = step_size
568  for h, elem_h in enumerate(parameter_names):
569  hdiff = center[elem_h]*step_size
570  if hdiff == 0.0:
571  hdiff = step_size
572  # Since the matrix is symmetric, we only compute the
573  # elements of one half of it plus the diagonal.
574  if k > h:
575  continue
576  if k != h:
577  fisher_matrix[k][h] = compute_fisher_element(
578  data, cosmo, center,
579  (elem_k, kdiff),
580  (elem_h, hdiff))
581  fisher_matrix[h][k] = fisher_matrix[k][h]
582  else:
583  fisher_matrix[k][k], gradient[k] = compute_fisher_element(
584  data, cosmo, center,
585  (elem_k, kdiff))

587  return fisher_matrix, gradient
