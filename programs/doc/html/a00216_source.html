523  #
524  # Regroup mcmc parameters by blocks of same speed
525  #
526  # This method divides all varying parameters from :attr:`mcmc_parameters`
527  # into as many categories as there are likelihoods, plus one (the slow
528  # block of cosmological parameters).
529  #
530  # It creates the attribute :attr:`block_parameters`, to be used in the
531  # module :mod:`mcmc`.
532  #
533  # .. note::
534  #
535  # It does not compute by any mean the real speed of each parameter,
536  # instead, every parameter belonging to the same likelihood will
537  # be considered as fast as its neighbour.
538  #
539  # .. warning::
540  #
541  # It assumes that the nuisance parameters are already written
542  # sequentially, and grouped together (not necessarily in the order
543  # described in :attr:`experiments`). If you mix up the different
544  # nuisance parameters in the .param file, this routine will not
545  # method as intended. It also assumes that the cosmological
546  # parameters are written at the beginning of the file.
547  #
548  #
549  def group_parameters_in_blocks(self):
550  array = []
551  # First obvious block is all cosmological parameters
552  array.append(len(self.get_mcmc_parameters(['varying', 'cosmo'])))
553  # Then, store all nuisance parameters
554  nuisance = self.get_mcmc_parameters(['varying', 'nuisance'])
555 
556  # Create an array to keep track of the already taken into account
557  # nuisance parameters. This will come in handy when using likelihoods
558  # that share some nuisance parameters.
559  used_nuisance = []
560  for likelihood in self.lkl.itervalues():
561  count = 0
562  for elem in nuisance:
563  if elem in likelihood.nuisance:
564  if elem not in used_nuisance:
565  count += 1
566  used_nuisance.append(elem)
567  likelihood.varying_nuisance_parameters = count
568 
569  # Then circle through them
570  index = 0
571  while index < len(nuisance):
572  elem = nuisance[index]
573  flag = False
574  # For each one, check if they belong to a likelihood
575  for likelihood in self.lkl.itervalues():
576  if elem in likelihood.nuisance:
577  # If yes, store the number of nuisance parameters needed
578  # for this likelihood.
579  flag = True
580  array.append(
581  likelihood.varying_nuisance_parameters+array[-1])
582  index += likelihood.varying_nuisance_parameters
583  continue
584  if not flag:
585  # If the loop reaches this part, it means this nuisance
586  # parameter was associated with no likelihood: this should not
587  # happen
588  raise io_mp.ConfigurationError(
589  "nuisance parameter %s " % elem +
590  "is associated to no likelihood")
591  # Store the result
592  self.block_parameters = array
593 
594  # Setting a default value for the over_sampling array
595  if not self.over_sampling:
596  self.over_sampling = [1 for _ in range(len(self.block_parameters))]
597  # Test that the over_sampling list has the same size as
598  # block_parameters.
599  else:
600  try:
601  assert len(self.block_parameters) == len(self.over_sampling)
602  except AssertionError:
603  raise io_mp.ConfigurationError(
604  "The length of the over_sampling field should be"
605  " equal to the number of blocks (one for cosmological "
606  "parameters, plus one for each likelihood with "
607  "nuisance parameters)")
608 
609  # Create a list of indices corresponding of the oversampling strategy
610  self.assign_over_sampling_indices()

724  #
725  # Put in :attr:`cosmo_arguments` the current values of
726  # :attr:`mcmc_parameters`
727  #
728  # This method is called at every step in the Markov chain, to update the
729  # dictionary. In the Markov chain, the scale is not remembered, so one
730  # has to apply it before giving it to the cosmological code.
731  #
732  # .. note::
733  #
734  # When you want to define new parameters in the Markov chain that do
735  # not have a one to one correspondance to a cosmological name, you
736  # can redefine its behaviour here. You will find in the source
737  # several such examples.
738  #
739  # .. note::
740  #
741  # For complex CLASS parameters, that expect a string of numbers
742  # separated with commas, you can now use the name of the argument,
743  # for instance :code:`m_ncdm`, then append a double underscore and a
744  # number. So if you run with two cosmological parameters,
745  # :code:`m_ncdm__1` and :code:`m_ncdm__2`, this function will
746  # automatically concatenate the two and feed class :code:`m_ncdm`.
747  # You still have to make sure that the other variables are properly
748  # set, like :code:`N_ncdm` to 2, in this example.
749  #
750  #
751  def update_cosmo_arguments(self):
752  # For all elements in any cosmological parameters
753  for elem in self.get_mcmc_parameters(['cosmo']):
754  # Fill in the dictionnary with the current value of parameters
755  self.cosmo_arguments[elem] = \
756  self.mcmc_parameters[elem]['current'] *\
757  self.mcmc_parameters[elem]['scale']
758 
759  # For all elements in the cosmological parameters from the mcmc list,
760  # translate any-one that is not directly a CLASS parameter into one.
761  # The try: except: syntax ensures that the first call
762  for elem in self.get_mcmc_parameters(['cosmo']):
763  # infer h from Omega_Lambda and delete Omega_Lambda
764  if elem == 'Omega_Lambda':
765  omega_b = self.cosmo_arguments['omega_b']
766  omega_cdm = self.cosmo_arguments['omega_cdm']
767  Omega_Lambda = self.cosmo_arguments['Omega_Lambda']
768  self.cosmo_arguments['h'] = math.sqrt(
769  (omega_b+omega_cdm) / (1.-Omega_Lambda))
770  del self.cosmo_arguments[elem]
771  # infer omega_cdm from Omega_L and delete Omega_L
772  elif elem == 'Omega_L':
773  omega_b = self.cosmo_arguments['omega_b']
774  h = self.cosmo_arguments['h']
775  Omega_L = self.cosmo_arguments['Omega_L']
776  self.cosmo_arguments['omega_cdm'] = (1.-Omega_L)*h*h-omega_b
777  del self.cosmo_arguments[elem]
778  elif elem == 'ln10^{10}A_s':
779  self.cosmo_arguments['A_s'] = math.exp(
780  self.cosmo_arguments[elem]) / 1.e10
781  del self.cosmo_arguments[elem]
782  elif elem == 'exp_m_2_tau_As':
783  tau_reio = self.cosmo_arguments['tau_reio']
784  self.cosmo_arguments['A_s'] = self.cosmo_arguments[elem] * \
785  math.exp(2.*tau_reio)
786  del self.cosmo_arguments[elem]
787  elif elem == 'f_cdi':
788  self.cosmo_arguments['n_cdi'] = self.cosmo_arguments['n_s']
789  elif elem == 'beta':
790  self.cosmo_arguments['alpha'] = 2.*self.cosmo_arguments['beta']
791  elif elem == 'M_tot':
792  self.cosmo_arguments['m_ncdm'] = self.cosmo_arguments['M_tot']/3.
793  del self.cosmo_arguments[elem]
794  # Finally, deal with all the parameters ending with __i, where i is
795  # an integer. Replace them all with their name without the trailing
796  # double underscore, concatenated with each other. The test is
797  # always on the one ending with __1, as it will be the first on the
798  # list, and deal with all the others.
799  elif re.search(r'__1', elem):
800  original_name = re.search(r'(.*)__1', elem).groups()[0]
801  # Recover the values of all the other elements
802  values = [self.cosmo_arguments[elem]]
803  for other_elem in self.get_mcmc_parameters(['cosmo']):
804  match = re.search(r'%s__([2-9])' % original_name,
805  other_elem)
806  if match:
807  values.append(self.cosmo_arguments[other_elem])
808  # create the cosmo_argument
809  self.cosmo_arguments[original_name] = ', '.join(
810  ['%g' % value for value in values])
811  # Delete the now obsolete entries of the dictionary
812  for index in range(1, len(values)+1):
813  del self.cosmo_arguments[
814  original_name + '__%i' % index]
