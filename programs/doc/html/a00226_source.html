148 def parse_docstring(docstring, key_symbol="<**>", description_symbol="<++>"):
149  # remove new lines and multiple whitespaces
150  whitespaces = re.compile(r"\s+")
151  docstring = whitespaces.sub(" ", docstring)
152 
153  # escape special characters
154  key_symbol = re.escape(key_symbol)
155  description_symbol = re.escape(description_symbol)
156 
157  # define the regular expressions to match the key and the description
158  key_match = r'{0}-{{0,2}}(.+?){0}'
159  re_key = re.compile(key_match.format(key_symbol))
160 
161  desc_match = r'({0}.+?{0}.+?{0})'
162  re_desc = re.compile(desc_match.format(description_symbol))
163 
164  # get all and check that the keys and descriptions have the same lenghts
165  keys = re_key.findall(docstring)
166  descriptions = re_desc.findall(docstring)
167  if len(keys) != len(descriptions):
168  msg = "The option keys and their descriptions have different lenghts.\n"
169  msg += "Make sure that there are as many string surrounded by '{0}'"
170  msg += " as there are surrounded by '{1}"
171  raise ValueError(msg.format(key_symbol, description_symbol))
172 
173  helpdict = dict(zip(keys, descriptions))
174  return helpdict

336 #
337 # Definition of the parser command line options
338 #
339 # The main parser has so far two subparsers, corresponding to the two main
340 # modes of operating the code, namely `run` and `info`. If you simply call
341 # :code:`python montepython/MontePython.py -h`, you will find only this piece
342 # of information. To go further, and find the command line options specific
343 # to these two submodes, one should then do: :code:`python
344 # montepython/MontePython.py run -h`, or :code:`info -h`.
345 #
346 # All command line arguments are defined below, for each of the two
347 # subparsers. This function create the automatic help command.
348 #
349 # Each flag outputs the following argument to a destination variable,
350 # specified by the `dest` keyword argument in the source code. Please check
351 # there to understand the variable names associated with each option.
352 #
353 # Options
354 # -------
355 #
356 # **run**
357 #
358 # <**>-N<**> : int
359 # <++>number of steps in the chain<++> (**OBL**). Note that when
360 # running on a cluster, your run might be stopped before reaching
361 # this number.<++>
362 # <**>-o<**> : str
363 # <++>output folder<++> (**OBL**). For instance :code:`-o
364 # chains/myexperiments/mymodel`. Note that in this example, the
365 # folder :code:`chains/myexperiments` must already exist.<++>
366 # <**>-p<**> : str
367 # <++>input parameter file<++> (**OBL**). For example :code:`-p
368 # input/exoticmodel.param`.<++>
369 # <**>-c<**> : str
370 # <++>input covariance matrix<++> (*OPT*). A covariance matrix is
371 # created when analyzing previous runs.
372 #
373 # Note that the list of parameters in the input covariance matrix and
374 # in the run do not necessarily coincide.<++>
375 #
376 # <**>-j<**> : str
377 # <++>jumping method<++> (`global` (default), `sequential` or `fast`)
378 # (*OPT*).
379 #
380 # With the `global` method the code generates a new random direction
381 # at each step, with the `sequential` one it cycles over the
382 # eigenvectors of the proposal density (= input covariance matrix).
383 #
384 # The `global` method the acceptance rate is usually lower but the
385 # points in the chains are less correlated. We recommend using the
386 # sequential method to get started in difficult cases, when the
387 # proposal density is very bad, in order to accumulate points and
388 # generate a covariance matrix to be used later with the `default`
389 # jumping method.
390 #
391 # The `fast` method implements the Cholesky decomposition presented
392 # in http://arxiv.org/abs/1304.4473 by Antony Lewis.<++>
393 # <**>-m<**> : str
394 # <++>sampling method<++>, by default 'MH' for Metropolis-Hastings,
395 # can be set to 'NS' for Nested Sampling (using Multinest wrapper
396 # PyMultiNest), 'CH' for Cosmo Hammer (using the Cosmo Hammer wrapper
397 # to emcee algorithm), and finally 'IS' for importance sampling.
398 #
399 # Note that when running with Importance sampling, you need to
400 # specify a folder to start from.<++>
401 # <**>-f<**> : float
402 # <++>jumping factor<++> (>= 0, default to 2.4) (*OPT*).
403 #
404 # The proposal density is given by the input covariance matrix (or a
405 # diagonal matrix with elements given by the square of the input
406 # sigma's) multiplied by the square of this factor. In other words, a
407 # typical jump will have an amplitude given by sigma times this
408 # factor.
409 #
410 # The default is the famous factor 2.4, found by **TO CHECK** Dunkley
411 # et al. to be an optimal trade-off between high acceptance rate and
412 # high correlation of chain elements, at least for multivariate
413 # gaussian posterior probabilities. It can be a good idea to reduce
414 # this factor for very non-gaussian posteriors.
415 #
416 # Using :code:`-f 0 -N 1` is a convenient way to get the likelihood
417 # exactly at the starting point passed in input.<++>
418 # <**>--conf<**> : str
419 # <++>configuration file<++> (default to `default.conf`) (*OPT*).
420 # This file contains the path to your cosmological module
421 # directory.<++>
422 # <**>--chain-number<**> : str
423 # arbitrary <++>number of the output chain<++>, to overcome the
424 # automatic one (*OPT*).
425 #
426 # By default, the chains are named :code:`yyyy-mm-dd_N__i.txt` with
427 # year, month and day being extracted, :code:`N` being the number of
428 # steps, and :code:`i` an automatically updated index.
429 #
430 # This means that running several times the code with the same
431 # command will create different chains automatically.
432 #
433 # This option is a way to enforce a particular number :code:`i`.
434 # This can be useful when running on a cluster: for instance you may
435 # ask your script to use the job number as :code:`i`.<++>
436 # <**>-r<**> : str
437 # <++>restart from last point in chain<++>, to avoid the burn-in
438 # stage (*OPT*).
439 #
440 # At the beginning of the run, the previous chain will be deleted,
441 # and its content transfered to the beginning of the new chain.<++>
442 # <**>-b<**> : str
443 # <++>start a new chain from the bestfit file<++> computed with
444 # analyze. (*OPT*)<++>
445 # <**>--fisher<**> : None
446 # <++>Calculates the inverse of the fisher matrix<++> to use as
447 # proposal distribution<++>
448 # <**>--silent<**> : None
449 # <++>silence the standard output<++> (useful when running on
450 # clusters)<++>
451 # <**>--Der-target-folder<**> : str
452 # <++>Add additional derived params to this folder<++>. It has to be
453 # used in conjunction with `Der-param-list`, and the method set to
454 # Der: :code:`-m Der`. (*OPT*)<++>
455 # <**>--Der-param-list<**> : str
456 # <++>Specify a number of derived parameters to be added<++>. A
457 # complete example would be to add Omega_Lambda as a derived
458 # parameter:
459 # :code:`python montepython/MontePython.py run -o existing_folder
460 # -m Der --Der-target-folder non_existing_folder --Der-param-list
461 # Omega_Lambda`<++>
462 # <**>--IS-starting-folder<**> : str
463 # <++>Perform Importance Sampling from this folder or set of
464 # chains<++> (*OPT*)<++>
465 #
466 # For Nested Sampling and Cosmo Hammer arguments, see
467 # :mod:`nested_sampling` and :mod:`cosmo_hammer`.
468 #
469 # **info**
470 #
471 # Replaces the old **-info** command, which is deprecated but still
472 # available.
473 #
474 # <**>files<**> : string/list of strings
475 # <++>You can specify either single files, or a complete folder<++>,
476 # for example :code:`info chains/my-run/2012-10-26*`, or :code:`info
477 # chains/my-run`.
478 #
479 # If you specify several folders (or set of files), a comparison
480 # will be performed.<++>
481 # <**>--bins<**> : int
482 # <++>number of bins in the histograms<++> used to derive posterior
483 # probabilities and credible intervals (default to 20). Decrease this
484 # number for smoother plots at the expense of masking details.<++>
485 # <**>--no-mean<**> : None
486 # <++>remove the mean likelihood from the plot<++>. By default, when
487 # plotting marginalised 1D posteriors, the code also shows the mean
488 # likelihood per bin with dashed lines; this flag switches off the
489 # dashed lines.<++>
490 # <**>--extra<**> : str
491 # <++>extra file to customize the output plots<++>. You can actually
492 # set all the possible options in this file, including line-width,
493 # ticknumber, ticksize, etc... You can specify four fields,
494 # `info.redefine` (dict with keys set to the previous variable, and
495 # the value set to a numerical computation that should replace this
496 # variable), `info.to_change` (dict with keys set to the old variable
497 # name, and value set to the new variable name), `info.to_plot` (list
498 # of variables with new names to plot), and `info.new_scales` (dict
499 # with keys set to the new variable names, and values set to the
500 # number by which it should be multiplied in the graph).<++> For
501 # instance,
502 #
503 # .. code::
504 #
505 # info.to_change={'oldname1':'newname1','oldname2':'newname2',...}
506 # info.to_plot=['name1','name2','newname3',...]
507 # info.new_scales={'name1':number1,'name2':number2,...}
508 #
509 # <**>--noplot<**> : bool
510 # <++>do not produce any plot, simply compute the posterior<++>

512 # <**>--noplot-2d<**> : bool
