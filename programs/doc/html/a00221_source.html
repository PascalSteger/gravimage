102 def calc_expvals(inpmod_l, inpmod_Cs_l, bpwf_l, bpwf_Cs_l):
103 
104  # Inputs
105  # inpmod: theory spectrum loaded by load_cmbfast (l, Cs_l)
106  # Contents: TT, TE, EE, BB, TB, EB, ET, BT, BE
107  # bpwf: bandpower window function from reduc_bpwf (l, Cs_l)
108  # Contents: TT, TP, EE->EE, BB->BB, EE->BB, BB->EE
109 
110  nbin = np.shape(bpwf_Cs_l)[1]
111 
112  # Don't assume inpmod and bpwf start at the same ell --
113  # CAMB spectra like to start at l=0 but bpwf can be higher.
114  # We do assume that both have delta ell = 1
115  nl = np.shape(bpwf_Cs_l)[0]
116  indx = np.arange(0,nl) # Python ranges want one more...
117  indx = indx + np.nonzero(bpwf_l[0]==inpmod_l)[0][0] # don't subtract 1
118 
119  # Initialize expval array
120  expv = np.zeros([nbin,np.shape(bpwf_Cs_l)[2]])
121 
122  # TT
123  x = bpwf_Cs_l[:,:,0]*np.transpose(np.tile(inpmod_Cs_l[indx,0],(nbin,1)))
124  expv[:,0] = np.sum(x,0)
125 
126  # TE
127  x = bpwf_Cs_l[:,:,1]*np.transpose(np.tile(inpmod_Cs_l[indx,1],(nbin,1)))
128  expv[:,1] = np.sum(x,0)
129 
130  # EE: x1 = EE->EE, x2 = BB->EE
131  x1 = bpwf_Cs_l[:,:,2]*np.transpose(np.tile(inpmod_Cs_l[indx,2],(nbin,1)))
132  x2 = bpwf_Cs_l[:,:,5]*np.transpose(np.tile(inpmod_Cs_l[indx,3],(nbin,1)))
133  expv[:,2] = np.sum(x1,0) + np.sum(x2,0)
134 
135  # BB: x1 = BB->BB, x2 = EE->BB
136  x1 = bpwf_Cs_l[:,:,3]*np.transpose(np.tile(inpmod_Cs_l[indx,3],(nbin,1)))
137  x2 = bpwf_Cs_l[:,:,4]*np.transpose(np.tile(inpmod_Cs_l[indx,2],(nbin,1)))
138  expv[:,3] = np.sum(x1,0) + np.sum(x2,0)
139 
140  # expv of TB, EB zero as initialized
141 
142  return expv

220 def calc_vecp(l,C_l_hat,C_fl, C_l):
221 
222  C_fl_12 = sqrtm(C_fl[l])
223  C_l_inv = LA.inv(C_l[l])
224  C_l_inv_12= sqrtm(C_l_inv)
225  # the order is inverted compared to matlab hamimeche_lewis_likelihood.m line 19
226 
227  # line 20 of hamimeche_lewis_likelihood.m
228  res = np.dot(C_l_inv_12, np.dot(C_l_hat[l], C_l_inv_12))
229  [d, u] = LA.eigh(res)
230  d = np.diag(d) # noticed that python returns the eigenvalues as a vector, not a matrix
231  #np. dot( u, np.dot( np.diag(d), LA.inv(u))) should be equals to res
232  # real symmetric matrices are diagnalized by orthogonal matrices (M^t M = 1)
233 
234  # this makes a diagonal matrix by applying g(x) to the eigenvalues, equation 10 in Barkats et al
235  gd = np.sign(np.diag(d) - 1) * np.sqrt(2 * (np.diag(d) - np.log(np.diag(d)) - 1))
236  gd = np.diag(gd)
237  # Argument of vecp in equation 8; multiplying from right to left
238  X = np.dot(np.transpose(u), C_fl_12)
239  X = np.dot(gd, X)
240  X = np.dot(u, X)
241  X = np.dot(C_fl_12, X)
242  # This is the vector of equation 7
243  X = vecp(X)
244 
245  return X

252 def vecp(mat):
253  # This returns the unique elements of a symmetric matrix
254  # 2014-02-11 now mirrors matlab vecp.m
255 
256  dim = mat.shape[0]
257 
258  vec = np.zeros((dim*(dim+1)/2))
259  counter = 0
260  for iDiag in range(0,dim):
261  vec[counter:counter+dim-iDiag] = np.diag(mat,iDiag)
262 
263  counter = counter + dim - iDiag
264 
265  return vec

268 # Function to evaluate the likelihood itself
269 def evaluateLikelihood(C_l,C_l_hat,C_fl,M_inv):
270  logL = 0
271  # Calculate X vector (Eq 8) for each l, lp
272  for l in range(0,9):
273  X = calc_vecp(l,C_l_hat,C_fl,C_l)
274  for lp in range(0,9):
275  #print l, lp, r
276  Xp = calc_vecp(lp,C_l_hat,C_fl,C_l)
277  M_inv_pp = M_inv[l,lp,:,:]
278  # calculate loglikelihood (Eq 7)
279  thislogL = (-0.5)*np.dot(X,np.dot(M_inv_pp,Xp))
280  logL = logL + thislogL
281 
282  if np.isnan(logL):
283  logL = -1e20
284 
285  logL = np.real(logL)
286  return logL
